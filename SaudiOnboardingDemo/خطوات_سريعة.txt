📝 لتغيير المدة (Duration):
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

افتح: LaunchScreenView.swift
Open: LaunchScreenView.swift

ابحث عن:
Find:

    DispatchQueue.main.asyncAfter(deadline: .now() + 2.0) {

غيّر 2.0 إلى:
Change 2.0 to:

    + 1.0  → ثانية واحدة / 1 second
    + 2.0  → ثانيتان (الحالي) / 2 seconds (current)
    + 3.0  → 3 ثوانٍ / 3 seconds
    + 5.0  → 5 ثوانٍ / 5 seconds

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Open: LaunchScreenView.swift

    250 → كبير / Large
    300 → كبير جداً / Extra Large

❌ المشكلة: الصورة لا تظهر
Problem: Image doesn't appear
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

✅ الحل:
Solution:

1. تأكد من اسم الصورة: launch_logo.png
   Check image name: launch_logo.png

2. تأكد من مكان الصورة الصحيح
   Check correct location

3. أعد بناء المشروع:
   Rebuild project:
   
   Cmd + Shift + K  (Clean)
   Cmd + R          (Run)


❌ المشكلة: اللوجو لا يظهر بوضوح
Problem: Logo not clear
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

✅ الحل:
Solution:

- استخدم صورة بدقة أعلى (400×400 أو أكبر)
  Use higher resolution (400×400 or larger)

- تأكد من لون اللوجو أبيض
  Make sure logo color is white

- تأكد من خلفية الصورة شفافة
  Make sure image background is transparent
